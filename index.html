function calculateQuote(wasteArea, stories, drainType, totalFeet) { let baseMaterialCost = wasteArea * 100; // Assuming $100 per waste area let heightHazard = (stories - 1) * 1000; // $1,000 per story over 1 let drainCost; switch (drainType) { case 'retrofit': drainCost = 750; // $750 per drain retrofit break; case 'replacement': drainCost = 1500; // $1,500 per drain replacement break; default: drainCost = 0; // No cost for other types } let gutterCost = totalFeet * 10; // $10 per foot of gutter let totalCost = baseMaterialCost + heightHazard + drainCost + gutterCost; return { totalCost: totalCost, itemizedBreakdown: { baseMaterialCost: baseMaterialCost, heightHazard: heightHazard, drainCost: drainCost, gutterCost: gutterCost } }; } // Example usage let quote = calculateQuote(50, 3, 'replacement', 20); console.log(quote); // Returns total cost and itemized breakdown
